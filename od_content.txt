function renderBills(){
  var bb=document.getElementById("billsTb");bb.innerHTML="";
  if(!curOwner){bb.innerHTML='<tr class="erow"><td colspan="5">Select your unit to see bills.</td></tr>';return;}
  var rows=maintenance.filter(function(m){return m.unit===curUnit;});
  if(!rows.length){bb.innerHTML='<tr class="erow"><td colspan="5">No bills found.</td></tr>';}
  else rows.forEach(function(m){var tr=document.createElement("tr");tr.innerHTML="<td>July 2026</td><td class='amt-cell'>"+fmt(m.amount)+"</td><td>"+esc(m.due)+"</td><td><span class='badge "+m.status+"'>"+slbl(m.status)+"</span></td><td>"+(m.status==="ok"?"<span style='color:var(--muted);font-size:.8rem;'>Paid</span>":"<button class='primary-btn teal' style='padding:6px 14px;font-size:.78rem;' onclick='openPay()'>Pay Now</button>")+"</td>";bb.appendChild(tr);});
  var hb=document.getElementById("histTb");hb.innerHTML="";
  var myP=recentPayments.filter(function(p){return p.unit===curUnit;});
  if(!myP.length){hb.innerHTML='<tr class="erow"><td colspan="4">'+(curOwner?"No payments yet.":"Select your unit.")+"</td></tr>";}
  else myP.forEach(function(p){var tr=document.createElement("tr");tr.innerHTML="<td>"+esc(p.date)+"</td><td class='amt-cell'>"+fmt(p.amount)+"</td><td>"+esc(p.mode)+"</td><td><span class='badge ok'>Confirmed</span></td>";hb.appendChild(tr);});
}

}
